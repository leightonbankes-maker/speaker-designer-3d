// --- Undo/Redo History ---
const MAX_UNDO = 50;
let undoStack = [];
let redoStack = [];

function captureState() {
  return {
    vertices: cubeVertices.map(v => v.clone()),
    edges: edges.map(e => [...e]),
    faces: faces.map(f => ({ verts: [...f.verts], name: f.name })),
    splits: new Set(splitVertices),
    subdivEdges: new Set(subdivisionEdges),
    ports: ports.map(p => ({ ...p })),
    wallThickness: WALL_THICKNESS,
  };
}

function restoreState(state) {
  cubeVertices = state.vertices.map(v => v.clone());
  edges = state.edges.map(e => [...e]);
  faces = state.faces.map(f => ({ verts: [...f.verts], name: f.name }));
  splitVertices = new Set(state.splits);
  subdivisionEdges = new Set(state.subdivEdges || []);
  ports = state.ports.map(p => ({ ...p }));
  WALL_THICKNESS = state.wallThickness;
  updateAll();
  updateTSPanel();
  refreshAllUI();
  if (typeof updateActionRow === 'function') updateActionRow();
}

function saveUndoState() {
  undoStack.push(captureState());
  if (undoStack.length > MAX_UNDO) undoStack.shift();
  redoStack = [];
  if (typeof updateActionRow === 'function') updateActionRow();
}

function undo() {
  if (undoStack.length === 0) return;
  redoStack.push(captureState());
  const prev = undoStack.pop();
  restoreState(prev);
}

function redo() {
  if (redoStack.length === 0) return;
  undoStack.push(captureState());
  const next = redoStack.pop();
  restoreState(next);
}

// --- Helpers ---
function clearGroup(group) {
  while (group.children.length > 0) {
    const child = group.children[0];
    group.remove(child);
    if (child.geometry) child.geometry.dispose();
    if (child.material) {
      if (Array.isArray(child.material)) child.material.forEach(m => m.dispose());
      else child.material.dispose();
    }
  }
}

function snapValue(val) {
  return Math.round(val / SNAP_GRID) * SNAP_GRID;
}

let edgeHitMeshes = [];
function buildEdgeHitMeshes() {
  clearGroup(edgeHitGroup);
  edgeHitMeshes = [];
  edges.forEach(([a, b], idx) => {
    const va = cubeVertices[a];
    const vb = cubeVertices[b];
    const mid = va.clone().add(vb).multiplyScalar(0.5);
    const len = va.distanceTo(vb);
    if (len < 0.5) return; // too short to split
    const cylGeom = new THREE.CylinderGeometry(EDGE_SPLIT_THRESHOLD, EDGE_SPLIT_THRESHOLD, len, 8, 1);
    const cylMat = new THREE.MeshBasicMaterial({ visible: false, side: THREE.DoubleSide });
    const cyl = new THREE.Mesh(cylGeom, cylMat);
    cyl.position.copy(mid);
    // Orient cylinder along edge
    const dir = vb.clone().sub(va).normalize();
    const up = new THREE.Vector3(0, 1, 0);
    const quat = new THREE.Quaternion().setFromUnitVectors(up, dir);
    cyl.setRotationFromQuaternion(quat);
    cyl.userData.edgeIndex = idx;
    cyl.userData.edgeVerts = [a, b];
    edgeHitGroup.add(cyl);
    edgeHitMeshes.push(cyl);
  });
}

// --- Edge Split Logic ---
function splitEdge(edgeIdx, splitPoint) {
  const [a, b] = edges[edgeIdx];
  const va = cubeVertices[a];
  const vb = cubeVertices[b];

  // Project splitPoint onto the edge line to get exact position
  const edgeDir = vb.clone().sub(va);
  const edgeLen = edgeDir.length();
  const edgeNorm = edgeDir.clone().normalize();
  const toPoint = splitPoint.clone().sub(va);
  let t = toPoint.dot(edgeNorm) / edgeLen;
  // Clamp t so new vertex isn't too close to endpoints (at least 15% from each end)
  t = THREE.MathUtils.clamp(t, 0.15, 0.85);

  const newPos = va.clone().add(edgeDir.clone().multiplyScalar(t));
  newPos.x = snapValue(newPos.x);
  newPos.y = snapValue(Math.max(0, newPos.y));
  newPos.z = snapValue(newPos.z);

  // Add new vertex
  const newIdx = cubeVertices.length;
  cubeVertices.push(newPos);
  splitVertices.add(newIdx);

  // Replace the old edge [a, b] with two new edges [a, newIdx] and [newIdx, b]
  // First, update subdivision edge indices since splice shifts everything after edgeIdx
  const updatedSubdivEdges = new Set();
  subdivisionEdges.forEach(sei => {
    if (sei === edgeIdx) {
      // The split edge itself was a subdivision edge — both halves inherit it
      updatedSubdivEdges.add(edgeIdx);
      updatedSubdivEdges.add(edgeIdx + 1);
    } else if (sei > edgeIdx) {
      updatedSubdivEdges.add(sei + 1); // shifted by one due to splice inserting an extra edge
    } else {
      updatedSubdivEdges.add(sei);
    }
  });
  subdivisionEdges = updatedSubdivEdges;

  edges.splice(edgeIdx, 1, [a, newIdx], [newIdx, b]);

  // Update faces: any face that has consecutive verts a, b (or b, a) needs newIdx inserted between them
  faces.forEach(face => {
    const verts = face.verts;
    const len = verts.length;
    for (let i = 0; i < len; i++) {
      const curr = verts[i];
      const next = verts[(i + 1) % len];
      if ((curr === a && next === b) || (curr === b && next === a)) {
        // Insert newIdx between curr and next
        const insertPos = (i + 1) % len;
        if (insertPos === 0) {
          // next is at index 0, so append at end
          verts.push(newIdx);
        } else {
          verts.splice(insertPos, 0, newIdx);
        }
        break; // each face only has this edge once
      }
    }
  });

  return newIdx;
}

// --- Face Subdivision Logic ---
// Given a face and two vertex indices that lie on it, split the face into two faces
// along a new edge between those vertices.
function subdivideFace(faceIndex, vertA, vertB) {
  const face = faces[faceIndex];
  const verts = face.verts;
  const idxA = verts.indexOf(vertA);
  const idxB = verts.indexOf(vertB);
  if (idxA === -1 || idxB === -1 || idxA === idxB) return false;

  // Ensure idxA < idxB for consistent ordering
  let iA = Math.min(idxA, idxB);
  let iB = Math.max(idxA, idxB);

  // The two vertex sets that form the two new faces
  // Face1: verts from iA to iB (inclusive)
  // Face2: verts from iB to iA (wrapping around, inclusive)
  const face1Verts = [];
  for (let i = iA; i <= iB; i++) {
    face1Verts.push(verts[i]);
  }

  const face2Verts = [];
  for (let i = iB; i < verts.length; i++) {
    face2Verts.push(verts[i]);
  }
  for (let i = 0; i <= iA; i++) {
    face2Verts.push(verts[i]);
  }

  // Each new face needs at least 3 vertices
  if (face1Verts.length < 3 || face2Verts.length < 3) return false;

  // Add the new edge (mark as subdivision/internal brace edge)
  const newEdgeIdx = edges.length;
  edges.push([vertA, vertB]);
  subdivisionEdges.add(newEdgeIdx);

  // Determine names for new faces
  const baseName = face.name;
  const newFace1 = { verts: face1Verts, name: baseName + '-A' };
  const newFace2 = { verts: face2Verts, name: baseName + '-B' };

  // Replace the original face with the two new ones
  faces.splice(faceIndex, 1, newFace1, newFace2);

  // Update port face indices - ports on the old face need reassignment
  ports.forEach(p => {
    if (p.faceIndex === faceIndex) {
      // Determine which new face the port center falls in
      const worldPos = faceUVToWorld_withOldBasis(faceIndex, p.u, p.v, verts);
      // Check which new face contains this point
      if (isPointInsideFaceVerts(newFace1.verts, worldPos)) {
        // stays at faceIndex (which is now newFace1)
        // Recalculate UV in new face basis
        const newUV = worldToFaceUV(faceIndex, worldPos);
        p.u = newUV.u;
        p.v = newUV.v;
      } else {
        // Move to faceIndex + 1 (newFace2)
        p.faceIndex = faceIndex + 1;
        const newUV = worldToFaceUV(faceIndex + 1, worldPos);
        p.u = newUV.u;
        p.v = newUV.v;
      }
    } else if (p.faceIndex > faceIndex) {
      // Shift face index since we inserted an extra face
      p.faceIndex += 1;
    }
  });

  return true;
}

// Helper: compute world position from UV using an arbitrary vert list (before face is changed)
function faceUVToWorld_withOldBasis(faceIndex, u, v, vertList) {
  // Use the same basis computation but with the given vert list
  const vArr = vertList.map(i => cubeVertices[i]);
  const normal = new THREE.Vector3();
  for (let i = 0; i < vArr.length; i++) {
    const curr = vArr[i];
    const next = vArr[(i + 1) % vArr.length];
    normal.x += (curr.y - next.y) * (curr.z + next.z);
    normal.y += (curr.z - next.z) * (curr.x + next.x);
    normal.z += (curr.x - next.x) * (curr.y + next.y);
  }
  if (normal.lengthSq() < 1e-10) {
    const e1 = vArr[1].clone().sub(vArr[0]);
    const e2 = vArr[vArr.length - 1].clone().sub(vArr[0]);
    normal.copy(e1.cross(e2));
  }
  normal.normalize();

  let bestLen = 0, bestU = null;
  for (let i = 0; i < vArr.length; i++) {
    const edge = vArr[(i + 1) % vArr.length].clone().sub(vArr[i]);
    const len = edge.length();
    if (len > bestLen) { bestLen = len; bestU = edge.clone().normalize(); }
  }
  if (!bestU) bestU = new THREE.Vector3(1, 0, 0);
  const uDir = bestU.clone();
  const vDir = new THREE.Vector3().crossVectors(normal, uDir).normalize();
  const projected = vArr.map(p => ({ u: p.dot(uDir), v: p.dot(vDir) }));
  let minU = Infinity, maxU = -Infinity, minV = Infinity, maxV = -Infinity;
  projected.forEach(p => {
    if (p.u < minU) minU = p.u; if (p.u > maxU) maxU = p.u;
    if (p.v < minV) minV = p.v; if (p.v > maxV) maxV = p.v;
  });
  const uLen = (maxU - minU) || 1;
  const vLen = (maxV - minV) || 1;
  const worldU = minU + u * uLen;
  const worldV = minV + v * vLen;
  return uDir.clone().multiplyScalar(worldU).add(vDir.clone().multiplyScalar(worldV));
}

// Check if a point lies inside a face defined by a vert index list
function isPointInsideFaceVerts(vertIdxList, point) {
  const vArr = vertIdxList.map(i => cubeVertices[i]);
  // Compute face normal
  const normal = new THREE.Vector3();
  for (let i = 0; i < vArr.length; i++) {
    const curr = vArr[i];
    const next = vArr[(i + 1) % vArr.length];
    normal.x += (curr.y - next.y) * (curr.z + next.z);
    normal.y += (curr.z - next.z) * (curr.x + next.x);
    normal.z += (curr.x - next.x) * (curr.y + next.y);
  }
  normal.normalize();
  // Pick projection plane
  const ax = Math.abs(normal.x), ay = Math.abs(normal.y), az = Math.abs(normal.z);
  let getU, getV;
  if (ax >= ay && ax >= az) { getU = p => p.y; getV = p => p.z; }
  else if (ay >= ax && ay >= az) { getU = p => p.x; getV = p => p.z; }
  else { getU = p => p.x; getV = p => p.y; }
  const pu = getU(point), pv = getV(point);
  const poly = vArr.map(p => ({ u: getU(p), v: getV(p) }));
  let inside = false;
  for (let i = 0, j = poly.length - 1; i < poly.length; j = i++) {
    const xi = poly[i].u, yi = poly[i].v;
    const xj = poly[j].u, yj = poly[j].v;
    if (((yi > pv) !== (yj > pv)) && (pu < (xj - xi) * (pv - yi) / (yj - yi) + xi)) inside = !inside;
  }
  return inside;
}

// Find which faces a vertex belongs to
function getFacesForVertex(vertIdx) {
  const result = [];
  faces.forEach((face, fi) => {
    if (face.verts.includes(vertIdx)) result.push(fi);
  });
  return result;
}

// Find vertices on a given face that are eligible as subdivision endpoints:
// They must be non-adjacent to the first vertex in the face's vert list.
function getSubdivEligibleVerts(faceIndex, firstVert) {
  const face = faces[faceIndex];
  const verts = face.verts;
  const idx = verts.indexOf(firstVert);
  if (idx === -1) return [];
  const n = verts.length;
  const eligible = [];
  for (let i = 0; i < n; i++) {
    // Skip self, and skip immediate neighbors (adjacent)
    if (i === idx) continue;
    if (i === (idx + 1) % n) continue;
    if (i === (idx - 1 + n) % n) continue;
    eligible.push(verts[i]);
  }
  return eligible;
}

// Get all faces that share two specific vertices (i.e., the face to subdivide)
function getSharedFace(vertA, vertB) {
  for (let fi = 0; fi < faces.length; fi++) {
    const verts = faces[fi].verts;
    if (verts.includes(vertA) && verts.includes(vertB)) return fi;
  }
  return -1;
}

// Reset subdivision mode
function cancelSubdivide() {
  subdivideMode = false;
  subdivFirstVertex = null;
  subdivEligibleVerts = [];
  if (subdivHighlightMesh) {
    scene.remove(subdivHighlightMesh);
    if (subdivHighlightMesh.geometry) subdivHighlightMesh.geometry.dispose();
    if (subdivHighlightMesh.material) subdivHighlightMesh.material.dispose();
    subdivHighlightMesh = null;
  }
  // Reset handle colors
  handles.forEach(h => {
    const isSplit = splitVertices.has(h.userData.vertexIndex);
    if (h.material && h.material.color) h.material.color.setHex(isSplit ? 0xff44ff : 0x00ccff);
    if (h.material && h.material.emissive) h.material.emissive.setHex(isSplit ? 0x661166 : 0x004466);
    h.scale.set(1, 1, 1);
  });
  updateSubdivideUI();
}

// Try to perform subdivision click
function trySubdivideClick(pxy) {
  if (!subdivideMode) return false;

  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const hits = raycaster.intersectObjects(handles);
  if (hits.length === 0) return false;

  const clickedHandle = hits[0].object;
  const vertIdx = clickedHandle.userData.vertexIndex;
  if (vertIdx === undefined) return false;

  if (subdivFirstVertex === null) {
    // First vertex selection - must be on a face with enough verts for subdivision
    // Check if any face containing this vertex has at least 4 verts (minimum for subdivision)
    const facesWithVert = getFacesForVertex(vertIdx);
    const eligibleFaces = facesWithVert.filter(fi => faces[fi].verts.length >= 4);
    if (eligibleFaces.length === 0) return false;

    subdivFirstVertex = vertIdx;

    // Compute all eligible second vertices across all eligible faces
    subdivEligibleVerts = [];
    eligibleFaces.forEach(fi => {
      const eligible = getSubdivEligibleVerts(fi, vertIdx);
      eligible.forEach(v => {
        if (!subdivEligibleVerts.includes(v)) subdivEligibleVerts.push(v);
      });
    });

    if (subdivEligibleVerts.length === 0) {
      subdivFirstVertex = null;
      return false;
    }

    // Highlight: make first vertex green, eligible verts yellow, others dim
    handles.forEach(h => {
      const vi = h.userData.vertexIndex;
      if (vi === vertIdx) {
        h.material.color.setHex(0x44ff44);
        h.material.emissive.setHex(0x22aa22);
        h.scale.set(1.5, 1.5, 1.5);
      } else if (subdivEligibleVerts.includes(vi)) {
        h.material.color.setHex(0xffff44);
        h.material.emissive.setHex(0xaaaa22);
        h.scale.set(1.3, 1.3, 1.3);
      } else {
        h.material.color.setHex(0x333333);
        h.material.emissive.setHex(0x111111);
        h.scale.set(0.7, 0.7, 0.7);
      }
    });

    updateSubdivideUI();
    return true;

  } else {
    // Second vertex selection
    if (!subdivEligibleVerts.includes(vertIdx)) {
      // If clicking first vertex again, deselect
      if (vertIdx === subdivFirstVertex) {
        subdivFirstVertex = null;
        subdivEligibleVerts = [];
        // Reset colors
        handles.forEach(h => {
          const isSplit = splitVertices.has(h.userData.vertexIndex);
          if (h.material && h.material.color) h.material.color.setHex(isSplit ? 0xff44ff : 0x00ccff);
          if (h.material && h.material.emissive) h.material.emissive.setHex(isSplit ? 0x661166 : 0x004466);
          h.scale.set(1, 1, 1);
        });
        updateSubdivideUI();
        return true;
      }
      return false;
    }

    // Find the shared face
    const faceIdx = getSharedFace(subdivFirstVertex, vertIdx);
    if (faceIdx === -1) return false;

    // Perform subdivision
    saveUndoState();
    const success = subdivideFace(faceIdx, subdivFirstVertex, vertIdx);
    if (success) {
      showSubdivideFlash(cubeVertices[subdivFirstVertex], cubeVertices[vertIdx]);
      // Exit subdivide mode after successful split
      subdivFirstVertex = null;
      subdivEligibleVerts = [];
      updateAll();
      updateTSPanel();
      updateSubdivideUI();
      return true;
    }
    return false;
  }
}

// Visual flash for subdivision
function showSubdivideFlash(posA, posB) {
  const mid = posA.clone().add(posB).multiplyScalar(0.5);
  const len = posA.distanceTo(posB);
  const dir = posB.clone().sub(posA).normalize();

  const flashGeom = new THREE.CylinderGeometry(0.15, 0.15, len, 8);
  const flashMat = new THREE.MeshBasicMaterial({ color: 0x44ff44, transparent: true, opacity: 0.9 });
  const flash = new THREE.Mesh(flashGeom, flashMat);
  flash.position.copy(mid);
  const up = new THREE.Vector3(0, 1, 0);
  const quat = new THREE.Quaternion().setFromUnitVectors(up, dir);
  flash.setRotationFromQuaternion(quat);
  scene.add(flash);

  let t = 0;
  const fadeOut = () => {
    t += 0.03;
    flash.scale.set(1 + t * 3, 1, 1 + t * 3);
    flashMat.opacity = 0.9 * (1 - t);
    if (t < 1) {
      requestAnimationFrame(fadeOut);
    } else {
      scene.remove(flash);
      flashGeom.dispose();
      flashMat.dispose();
    }
  };
  fadeOut();
}

function updateSubdividePreview(pxy) {
  if (!subdivideMode || subdivFirstVertex === null) {
    if (subdivHighlightMesh) {
      scene.remove(subdivHighlightMesh);
      if (subdivHighlightMesh.geometry) subdivHighlightMesh.geometry.dispose();
      if (subdivHighlightMesh.material) subdivHighlightMesh.material.dispose();
      subdivHighlightMesh = null;
    }
    return;
  }

  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const hits = raycaster.intersectObjects(handles);

  let targetPos = null;
  let validTarget = false;

  if (hits.length > 0) {

    if (subdivEligibleVerts.includes(vi)) {
      targetPos = cubeVertices[vi].clone();
      validTarget = true;
    }
  }

  if (subdivHighlightMesh) {
    scene.remove(subdivHighlightMesh);
    if (subdivHighlightMesh.geometry) subdivHighlightMesh.geometry.dispose();
    if (subdivHighlightMesh.material) subdivHighlightMesh.material.dispose();

    const geom = new THREE.BufferGeometry().setFromPoints([startPos.clone(), targetPos]);

    subdivHighlightMesh = new THREE.Line(geom, mat);
    subdivHighlightMesh.name = 'subdivPreviewLine';
    scene.add(subdivHighlightMesh);
  }
}

// --- Face Hit Meshes ---
let faceHitMeshes = [];
function buildFaceHitMeshes() {
  faceHitMeshes.forEach(m => {
    scene.remove(m);
    if (m.geometry) m.geometry.dispose();
    if (m.material) m.material.dispose();
  });
  faceHitMeshes = [];
  faces.forEach((face, fi) => {
    const v = face.verts.map(i => cubeVertices[i]);
    const normal = getFaceNormal(fi);
    const tris = triangulateFace(v, normal);
    const positions = [];
    tris.forEach(([a, b, c]) => {
      positions.push(v[a].x, v[a].y, v[a].z);
      positions.push(v[b].x, v[b].y, v[b].z);
      positions.push(v[c].x, v[c].y, v[c].z);
    });
    if (positions.length === 0) {
      for (let i = 1; i < v.length - 1; i++) {
        positions.push(v[0].x, v[0].y, v[0].z);
        positions.push(v[i].x, v[i].y, v[i].z);
        positions.push(v[i+1].x, v[i+1].y, v[i+1].z);
      }
    }
    const geom = new THREE.BufferGeometry();
    geom.setAttribute("position", new THREE.BufferAttribute(new Float32Array(positions), 3));
    geom.computeVertexNormals();
    const mat = new THREE.MeshBasicMaterial({ visible: false, side: THREE.DoubleSide });
    const mesh = new THREE.Mesh(geom, mat);
    mesh.userData.faceIndex = fi;
    faceHitMeshes.push(mesh);
    scene.add(mesh);
  });
}

function getFaceNormal(faceIndex) {
  const face = faces[faceIndex];
  const v = face.verts.map(i => cubeVertices[i]);
  // Use Newell's method for robust normal of N-gon
  const normal = new THREE.Vector3();
  for (let i = 0; i < v.length; i++) {
    const curr = v[i];
    const next = v[(i + 1) % v.length];
    normal.x += (curr.y - next.y) * (curr.z + next.z);
    normal.y += (curr.z - next.z) * (curr.x + next.x);
    normal.z += (curr.x - next.x) * (curr.y + next.y);
  }
  if (normal.lengthSq() < 1e-10) {
    // Fallback: use first three non-collinear verts
    const e1 = v[1].clone().sub(v[0]);
    const e2 = v[v.length - 1].clone().sub(v[0]);
    normal.copy(e1.cross(e2));
  }
  return normal.normalize();
}

// Compute a robust 2D coordinate system for any N-gon face.
// Returns origin, uAxis, vAxis (both unit vectors in the face plane),
// and uLen/vLen (the bounding extent of the face projected onto these axes).
// Also returns min2D so we can properly map 0..1 UV across the full face.
function getFaceBasis(faceIndex) {
  const face = faces[faceIndex];
  const v = face.verts.map(i => cubeVertices[i]);
  const normal = getFaceNormal(faceIndex);

  // Pick the longest edge as the U axis for stability
  let bestLen = 0;
  let bestU = null;
  for (let i = 0; i < v.length; i++) {
    const edge = v[(i + 1) % v.length].clone().sub(v[i]);
    const len = edge.length();
    if (len > bestLen) {
      bestLen = len;
      bestU = edge.clone().normalize();
    }
  }
  if (!bestU || bestLen < 1e-6) bestU = new THREE.Vector3(1, 0, 0);

  // V axis is perpendicular to U in the face plane
  const uDir = bestU.clone();
  const vDir = new THREE.Vector3().crossVectors(normal, uDir).normalize();

  // Project all vertices onto this 2D system to find bounding box
  const projected = v.map(p => ({ u: p.dot(uDir), v: p.dot(vDir) }));
  let minU = Infinity, maxU = -Infinity, minV = Infinity, maxV = -Infinity;
  projected.forEach(p => {
    if (p.u < minU) minU = p.u;
    if (p.u > maxU) maxU = p.u;
    if (p.v < minV) minV = p.v;
    if (p.v > maxV) maxV = p.v;
  });

  const uLen = maxU - minU;
  const vLen = maxV - minV;

  // Origin in 3D is the point corresponding to (minU, minV)
  const origin = uDir.clone().multiplyScalar(minU).add(vDir.clone().multiplyScalar(minV));

  return {
    origin,
    uDir, vDir, normal,
    uLen: uLen || 1,
    vLen: vLen || 1,
    minU, minV, maxU, maxV,
  };
}

function worldToFaceUV(faceIndex, worldPoint) {
  const basis = getFaceBasis(faceIndex);
  const pu = worldPoint.dot(basis.uDir);
  const pv = worldPoint.dot(basis.vDir);
  return {
    u: (pu - basis.minU) / basis.uLen,
    v: (pv - basis.minV) / basis.vLen,
  };
}

function faceUVToWorld(faceIndex, u, v) {
  const basis = getFaceBasis(faceIndex);
  const worldU = basis.minU + u * basis.uLen;
  const worldV = basis.minV + v * basis.vLen;
  return basis.uDir.clone().multiplyScalar(worldU).add(basis.vDir.clone().multiplyScalar(worldV));
}

function getOutwardNormal(faceIndex) {
  const normal = getFaceNormal(faceIndex);
  const face = faces[faceIndex];
  const v = face.verts.map(i => cubeVertices[i]);
  const faceCenter = new THREE.Vector3();
  v.forEach(vv => faceCenter.add(vv));
  faceCenter.divideScalar(v.length);
  const boxCenter = new THREE.Vector3();
  cubeVertices.forEach(cv => boxCenter.add(cv));
  boxCenter.divideScalar(cubeVertices.length);
  if (normal.dot(faceCenter.clone().sub(boxCenter)) < 0) normal.negate();
  return normal;
}

function getInwardNormal(faceIndex) {
  return getOutwardNormal(faceIndex).negate();
}

// --- Build Enclosure ---
// Ear-clipping triangulation for N-gon faces (handles convex and mildly concave polygons)
function triangulateFace(face3DVerts, faceNormal) {
  if (face3DVerts.length < 3) return [];
  if (face3DVerts.length === 3) return [[0, 1, 2]];

  // Project to 2D using the face normal to pick the dominant plane
  const ax = Math.abs(faceNormal.x);
  const ay = Math.abs(faceNormal.y);
  const az = Math.abs(faceNormal.z);
  let getU, getV;
  if (ax >= ay && ax >= az) {
    getU = (p) => p.y; getV = (p) => p.z;
  } else if (ay >= ax && ay >= az) {
    getU = (p) => p.x; getV = (p) => p.z;
  } else {
    getU = (p) => p.x; getV = (p) => p.y;
  }

  const pts2D = face3DVerts.map(p => ({ u: getU(p), v: getV(p) }));

  // Ensure CCW winding
  let area = 0;
  for (let i = 0; i < pts2D.length; i++) {
    const j = (i + 1) % pts2D.length;
    area += pts2D[i].u * pts2D[j].v;
    area -= pts2D[j].u * pts2D[i].v;
  }
  const indices = [];
  for (let i = 0; i < pts2D.length; i++) indices.push(i);
  if (area < 0) indices.reverse();

  const triangles = [];
  const remaining = [...indices];

  function isEar(prev, curr, next) {
    const a = pts2D[prev], b = pts2D[curr], c = pts2D[next];
    // Check if this triangle is CCW (convex)
    const cross = (b.u - a.u) * (c.v - a.v) - (b.v - a.v) * (c.u - a.u);
    if (cross <= 0) return false;
    // Check no other vertex is inside this triangle
    for (let i = 0; i < remaining.length; i++) {
      const pi = remaining[i];
      if (pi === prev || pi === curr || pi === next) continue;
      if (pointInTriangle2D(pts2D[pi], a, b, c)) return false;
    }
    return true;
  }

  let safety = remaining.length * 3;
  while (remaining.length > 3 && safety-- > 0) {
    let earFound = false;
    for (let i = 0; i < remaining.length; i++) {
      const prev = remaining[(i + remaining.length - 1) % remaining.length];
      const curr = remaining[i];
      const next = remaining[(i + 1) % remaining.length];
      if (isEar(prev, curr, next)) {
        triangles.push([prev, curr, next]);
        remaining.splice(i, 1);
        earFound = true;
        break;
      }
    }
    if (!earFound) break; // fallback: can't find ear, use fan
  }
  if (remaining.length === 3) {
    triangles.push([remaining[0], remaining[1], remaining[2]]);
  } else if (remaining.length > 3) {
    // Fallback to fan triangulation
    for (let i = 1; i < remaining.length - 1; i++) {
      triangles.push([remaining[0], remaining[i], remaining[i + 1]]);
    }
  }
  return triangles;
}

function pointInTriangle2D(p, a, b, c) {
  const d1 = (p.u - c.u) * (a.v - c.v) - (a.u - c.u) * (p.v - c.v);
  const d2 = (p.u - a.u) * (b.v - a.v) - (b.u - a.u) * (p.v - a.v);
  const d3 = (p.u - b.u) * (c.v - b.v) - (c.u - b.u) * (p.v - b.v);
  const hasNeg = (d1 < 0) || (d2 < 0) || (d3 < 0);
  const hasPos = (d1 > 0) || (d2 > 0) || (d3 > 0);
  return !(hasNeg && hasPos);
}

function buildEnclosure() {
  clearGroup(enclosureGroup);
  clearGroup(edgeGroup);
  faces.forEach((face, fi) => {
    const v = face.verts.map(i => cubeVertices[i]);
    const normal = getFaceNormal(fi);
    const tris = triangulateFace(v, normal);
    const positions = [];
    tris.forEach(([a, b, c]) => {
      positions.push(v[a].x, v[a].y, v[a].z);
      positions.push(v[b].x, v[b].y, v[b].z);
      positions.push(v[c].x, v[c].y, v[c].z);
    });
    if (positions.length === 0) {
      // Fallback: fan triangulation
      for (let i = 1; i < v.length - 1; i++) {
        positions.push(v[0].x, v[0].y, v[0].z);
        positions.push(v[i].x, v[i].y, v[i].z);
        positions.push(v[i+1].x, v[i+1].y, v[i+1].z);
      }
    }
    const geom = new THREE.BufferGeometry();
    geom.setAttribute("position", new THREE.BufferAttribute(new Float32Array(positions), 3));
    geom.computeVertexNormals();
    const panel = new THREE.Mesh(geom, panelMaterial);
    panel.castShadow = true; panel.receiveShadow = true;
    panel.name = `panel_${fi}`;
    enclosureGroup.add(panel);
    const wirePanel = new THREE.Mesh(geom.clone(), panelWireMaterial);
    wirePanel.name = `panelWire_${fi}`;
    enclosureGroup.add(wirePanel);
  });
  edges.forEach(([a, b], idx) => {
    const geom = new THREE.BufferGeometry().setFromPoints([cubeVertices[a].clone(), cubeVertices[b].clone()]);
    const isSubdivEdge = subdivisionEdges.has(idx);
    const mat = isSubdivEdge ? braceEdgeMaterial : edgeMaterial;
    const line = new THREE.Line(geom, mat);
    line.name = `edge_${idx}`;
    edgeGroup.add(line);
  });
  buildFaceHitMeshes();
  buildEdgeHitMeshes();
  buildBraces();
  buildCrosshatch();
  buildPortVisuals();
}

// --- Internal Brace Panels (3D slabs at subdivision edges) ---
function buildBraces() {
  clearGroup(braceGroup);

  subdivisionEdges.forEach(edgeIdx => {
    if (edgeIdx >= edges.length) return;
    const [a, b] = edges[edgeIdx];
    const va = cubeVertices[a];
    const vb = cubeVertices[b];
    const edgeLen = va.distanceTo(vb);
    if (edgeLen < 0.1) return;

    // Find the two faces that share this edge
    const adjacentFaces = [];
    faces.forEach((face, fi) => {
      const verts = face.verts;
      const hasA = verts.includes(a);
      const hasB = verts.includes(b);
      if (hasA && hasB) adjacentFaces.push(fi);
    });
    if (adjacentFaces.length < 2) return;

    // Compute the inward normal — average of the two face inward normals
    // This gives the direction the brace panel extends into the box interior
    const inNorm0 = getOutwardNormal(adjacentFaces[0]).negate();
    const inNorm1 = getOutwardNormal(adjacentFaces[1]).negate();
    // The brace should extend perpendicular to the edge along the interior
    // Use the face normals to determine the interior direction

    // Edge direction
    const edgeDir = vb.clone().sub(va).normalize();

    // The brace plane normal is perpendicular to the edge and lies in the interior
    // Compute it from the two face normals
    const avgInward = inNorm0.clone().add(inNorm1).normalize();

    // The brace extends perpendicular to the edge, into the interior
    // Project avgInward onto the plane perpendicular to the edge
    const braceDir = avgInward.clone().sub(edgeDir.clone().multiplyScalar(avgInward.dot(edgeDir))).normalize();

    if (braceDir.lengthSq() < 0.01) return;

    // Calculate how deep the brace should extend — use the enclosure interior depth
    // Raycast from edge midpoint inward to find the opposite wall
    const edgeMid = va.clone().add(vb).multiplyScalar(0.5);
    const braceDepth = calcBraceDepth(edgeMid, braceDir, edgeIdx);
    if (braceDepth < WALL_THICKNESS * 2) return; // too shallow

    // Actual brace depth = full interior depth minus wall thickness on the far side
    const actualDepth = braceDepth - WALL_THICKNESS;
    if (actualDepth < 0.5) return;

    // Build a 3D box (slab) for the brace:
    // - along the edge (length = edgeLen)
    // - into the interior (depth = actualDepth)
    // - thickness = WALL_THICKNESS

    // The slab's local coordinate system:
    // X = along edge, Y = braceDir (into interior), Z = perpendicular (thickness)
    const thicknessDir = new THREE.Vector3().crossVectors(edgeDir, braceDir).normalize();

    // 8 corners of the brace slab
    const halfThick = WALL_THICKNESS / 2;
    const corners = [];
    for (let xi = 0; xi <= 1; xi++) {
      for (let yi = 0; yi <= 1; yi++) {
        for (let zi = 0; zi <= 1; zi++) {
          const p = va.clone()
            .add(edgeDir.clone().multiplyScalar(xi * edgeLen))
            .add(braceDir.clone().multiplyScalar(yi * actualDepth))
            .add(thicknessDir.clone().multiplyScalar((zi - 0.5) * WALL_THICKNESS));
          corners.push(p);
        }
      }
    }

    // corners indices: [0]=va+0+0-ht, [1]=va+0+0+ht, [2]=va+0+d-ht, [3]=va+0+d+ht
    //                  [4]=vb+0+0-ht, [5]=vb+0+0+ht, [6]=vb+0+d-ht, [7]=vb+0+d+ht

    // 6 faces of the box
    const boxFaces = [
      [0, 4, 5, 1], // front (near edge)
      [2, 3, 7, 6], // back (far edge)
      [0, 1, 3, 2], // left (va side)
      [4, 6, 7, 5], // right (vb side)
      [1, 5, 7, 3], // top (+thickness)
      [0, 2, 6, 4], // bottom (-thickness)
    ];

    const positions = [];
    boxFaces.forEach(face => {
      // Two triangles per quad
      const c = face.map(i => corners[i]);
      positions.push(c[0].x, c[0].y, c[0].z, c[1].x, c[1].y, c[1].z, c[2].x, c[2].y, c[2].z);
      positions.push(c[0].x, c[0].y, c[0].z, c[2].x, c[2].y, c[2].z, c[3].x, c[3].y, c[3].z);
    });

    const geom = new THREE.BufferGeometry();
    geom.setAttribute("position", new THREE.BufferAttribute(new Float32Array(positions), 3));
    geom.computeVertexNormals();
    const mesh = new THREE.Mesh(geom, braceMaterial);
    mesh.castShadow = true;
    mesh.receiveShadow = true;
    mesh.name = `brace_${edgeIdx}`;
    braceGroup.add(mesh);

    // Outline edges for the brace
    const outlinePoints = [
      corners[0], corners[4], corners[5], corners[1], corners[0], // front face loop
      corners[2], corners[6], corners[7], corners[3], corners[2], // back face loop
    ];
    const lineGeom = new THREE.BufferGeometry().setFromPoints(outlinePoints);
    const line = new THREE.Line(lineGeom, braceEdgeMaterial);
    line.name = `braceEdge_${edgeIdx}`;
    braceGroup.add(line);

    // Connect front to back
    const connectors = [[0,2],[1,3],[4,6],[5,7]];
    connectors.forEach(([ci, cj]) => {
      const cGeom = new THREE.BufferGeometry().setFromPoints([corners[ci], corners[cj]]);
      const cLine = new THREE.Line(cGeom, braceEdgeMaterial);
      cLine.name = `braceConn_${edgeIdx}_${ci}_${cj}`;
      braceGroup.add(cLine);
    });

    // Brace label
    const labelPos = edgeMid.clone()
      .add(braceDir.clone().multiplyScalar(actualDepth * 0.5))
      .add(thicknessDir.clone().multiplyScalar(halfThick + 0.3));
    const canvas = document.createElement("canvas");
    canvas.width = 200; canvas.height = 48;
    const ctx = canvas.getContext("2d");
    ctx.fillStyle = "rgba(0,0,0,0.6)";
    ctx.beginPath();
    if (ctx.roundRect) ctx.roundRect(4, 4, 192, 40, 8); else ctx.rect(4, 4, 192, 40);
    ctx.fill();
    ctx.fillStyle = "#44ff44";
    ctx.font = "bold 16px Arial";
    ctx.textAlign = "center";
    ctx.fillText(`BRACE ${fmtLen(actualDepth)}×${fmtThickness(WALL_THICKNESS)}`, 100, 30);
    const texture = new THREE.CanvasTexture(canvas);
    const spriteMat = new THREE.SpriteMaterial({ map: texture, transparent: true, depthTest: false });
    const sprite = new THREE.Sprite(spriteMat);
    sprite.position.copy(labelPos);
    sprite.scale.set(3.5, 1.0, 1);
    sprite.name = `braceLabel_${edgeIdx}`;
    braceGroup.add(sprite);
  });
}

// Calculate how deep a brace can extend from a point along a direction
// by checking intersection with opposite enclosure faces
function calcBraceDepth(origin, direction, excludeEdgeIdx) {
  // Find the two faces adjacent to this subdivision edge
  const [ea, eb] = edges[excludeEdgeIdx];
  const adjacentFaceIndices = [];
  faces.forEach((face, fi) => {
    if (face.verts.includes(ea) && face.verts.includes(eb)) {
      adjacentFaceIndices.push(fi);
    }
  });

  let minDist = 100; // large default
  faces.forEach((face, fi) => {
    // Skip the two faces that share this edge
    if (adjacentFaceIndices.includes(fi)) return;

    const faceVerts = face.verts.map(i => cubeVertices[i]);
    const normal = getFaceNormal(fi);

    // Check if the ray from origin along direction intersects this face's plane
    const denom = normal.dot(direction);
    if (Math.abs(denom) < 1e-6) return; // parallel

    const t = normal.dot(faceVerts[0].clone().sub(origin)) / denom;
    if (t <= 0) return; // behind origin

    // Check if intersection point is inside the face polygon
    const hitPoint = origin.clone().add(direction.clone().multiplyScalar(t));
    if (isPointInsideFaceVerts(face.verts, hitPoint)) {
      if (t < minDist) minDist = t;
    }
  });

  return minDist;
}

// Calculate total volume of all internal braces
function calculateBraceVolume() {
  let totalVol = 0;
  subdivisionEdges.forEach(edgeIdx => {
    if (edgeIdx >= edges.length) return;
    const [a, b] = edges[edgeIdx];
    const va = cubeVertices[a];
    const vb = cubeVertices[b];
    const edgeLen = va.distanceTo(vb);
    if (edgeLen < 0.1) return;

    const adjacentFaces = [];
    faces.forEach((face, fi) => {
      if (face.verts.includes(a) && face.verts.includes(b)) adjacentFaces.push(fi);
    });
    if (adjacentFaces.length < 2) return;

    const inNorm0 = getOutwardNormal(adjacentFaces[0]).negate();
    const inNorm1 = getOutwardNormal(adjacentFaces[1]).negate();
    const edgeDir = vb.clone().sub(va).normalize();
    const avgInward = inNorm0.clone().add(inNorm1).normalize();
    const braceDir = avgInward.clone().sub(edgeDir.clone().multiplyScalar(avgInward.dot(edgeDir))).normalize();
    if (braceDir.lengthSq() < 0.01) return;

    const edgeMid = va.clone().add(vb).multiplyScalar(0.5);
    const braceDepth = calcBraceDepth(edgeMid, braceDir, edgeIdx);
    if (braceDepth < WALL_THICKNESS * 2) return;
    const actualDepth = braceDepth - WALL_THICKNESS;
    if (actualDepth < 0.5) return;

    totalVol += edgeLen * actualDepth * WALL_THICKNESS;
  });
  return totalVol;
}

// --- Front Face Cross-Hatching ---
const crosshatchMaterial = new THREE.LineBasicMaterial({ color: 0xff8866, transparent: true, opacity: 0.45, depthTest: true });

function buildCrosshatch() {
  clearGroup(crosshatchGroup);

  // Front face (index 0) — use the robust basis system for N-gon support
  const face = faces[0];
  const basis = getFaceBasis(0);
  const outNormal = getOutwardNormal(0);
  const offset = outNormal.clone().multiplyScalar(0.05);

  const uLen = basis.uLen;
  const vLen = basis.vLen;

  // Helper: convert local (u,v) in [0..uLen, 0..vLen] to world 3D + offset
  function localToWorld3D(lu, lv) {
    const wu = basis.minU + lu;
    const wv = basis.minV + lv;
    return basis.uDir.clone().multiplyScalar(wu).add(basis.vDir.clone().multiplyScalar(wv)).add(offset);
  }

  // Build a 2D polygon of the face for line clipping
  const faceVerts2D = face.verts.map(i => {
    const p = cubeVertices[i];
    return { u: p.dot(basis.uDir) - basis.minU, v: p.dot(basis.vDir) - basis.minV };
  });

  // Sutherland-Hodgman line-segment clip against convex/concave polygon
  // Clips a line segment (p1,p2) to the polygon, returns clipped segments
  function clipLineToPolygon(p1u, p1v, p2u, p2v, poly) {
    let segments = [{ u1: p1u, v1: p1v, u2: p2u, v2: p2v }];
    const n = poly.length;
    for (let i = 0; i < n; i++) {
      const eu = poly[(i+1)%n].u - poly[i].u;
      const ev = poly[(i+1)%n].v - poly[i].v;
      // Edge normal (inward for CCW polygon)
      const nu = -ev, nv = eu;
      const newSegments = [];
      segments.forEach(seg => {
        const d1 = nu * (seg.u1 - poly[i].u) + nv * (seg.v1 - poly[i].v);
        const d2 = nu * (seg.u2 - poly[i].u) + nv * (seg.v2 - poly[i].v);
        if (d1 >= -0.01 && d2 >= -0.01) {
          newSegments.push(seg); // both inside
        } else if (d1 >= -0.01) {
          // p1 inside, p2 outside — clip p2
          const t = d1 / (d1 - d2);
          newSegments.push({ u1: seg.u1, v1: seg.v1, u2: seg.u1 + t*(seg.u2-seg.u1), v2: seg.v1 + t*(seg.v2-seg.v1) });
        } else if (d2 >= -0.01) {
          // p2 inside, p1 outside — clip p1
          const t = d1 / (d1 - d2);
          newSegments.push({ u1: seg.u1 + t*(seg.u2-seg.u1), v1: seg.v1 + t*(seg.v2-seg.v1), u2: seg.u2, v2: seg.v2 });
        }
        // else both outside — discard
      });
      segments = newSegments;
      if (segments.length === 0) break;
    }
    return segments;
  }

  // Check polygon winding — ensure it's CCW for clipping
  let windingSum = 0;
  for (let i = 0; i < faceVerts2D.length; i++) {
    const curr = faceVerts2D[i];
    const next = faceVerts2D[(i+1)%faceVerts2D.length];
    windingSum += (next.u - curr.u) * (next.v + curr.v);
  }
  const clipPoly = windingSum > 0 ? [...faceVerts2D].reverse() : faceVerts2D;

  const spacing = 1.2;
  const numU = Math.ceil(uLen / spacing);
  const numV = Math.ceil(vLen / spacing);

  // Diagonal set 1
  for (let i = 1; i < numU + numV; i++) {
    const t = i * spacing;
    let startU, startV, endU, endV;
    if (t <= uLen) { startU = t; startV = 0; }
    else { startU = uLen; startV = t - uLen; }
    if (t <= vLen) { endU = 0; endV = t; }
    else { endU = t - vLen; endV = vLen; }
    if (startU < endU || startV > endV) continue;
    if (startU < 0 || startU > uLen || endU < 0 || endU > uLen) continue;
    if (startV < 0 || startV > vLen || endV < 0 || endV > vLen) continue;
    const clipped = clipLineToPolygon(startU, startV, endU, endV, clipPoly);
    clipped.forEach((seg, si) => {
      const p1 = localToWorld3D(seg.u1, seg.v1);
      const p2 = localToWorld3D(seg.u2, seg.v2);
      if (p1.distanceTo(p2) < 0.1) return;
      const geom = new THREE.BufferGeometry().setFromPoints([p1, p2]);
      const line = new THREE.Line(geom, crosshatchMaterial);
      line.name = `crosshatch_a_${i}_${si}`;
      crosshatchGroup.add(line);
    });
  }

  // Diagonal set 2
  for (let i = 1; i < numU + numV; i++) {
    const t = i * spacing;
    let startU, startV, endU, endV;
    if (t <= uLen) { startU = uLen - t; startV = 0; }
    else { startU = 0; startV = t - uLen; }
    if (t <= vLen) { endU = uLen; endV = t; }
    else { endU = uLen - (t - vLen); endV = vLen; }
    if (startU > endU || startV > endV) continue;
    if (startU < 0 || startU > uLen || endU < 0 || endU > uLen) continue;
    if (startV < 0 || startV > vLen || endV < 0 || endV > vLen) continue;
    const clipped = clipLineToPolygon(startU, startV, endU, endV, clipPoly);
    clipped.forEach((seg, si) => {
      const p1 = localToWorld3D(seg.u1, seg.v1);
      const p2 = localToWorld3D(seg.u2, seg.v2);
      if (p1.distanceTo(p2) < 0.1) return;
      const geom = new THREE.BufferGeometry().setFromPoints([p1, p2]);
      const line = new THREE.Line(geom, crosshatchMaterial);
      line.name = `crosshatch_b_${i}_${si}`;
      crosshatchGroup.add(line);
    });
  }

  // FRONT label
  const frontVerts = face.verts.map(i => cubeVertices[i]);
  const faceCenter = new THREE.Vector3();
  frontVerts.forEach(vv => faceCenter.add(vv));
  faceCenter.divideScalar(frontVerts.length);
  const labelPos = faceCenter.clone().add(outNormal.clone().multiplyScalar(0.15));
  const canvas = document.createElement("canvas");
  canvas.width = 256; canvas.height = 80;
  const ctx = canvas.getContext("2d");
  ctx.fillStyle = "rgba(0,0,0,0.55)";
  ctx.beginPath();
  if (ctx.roundRect) ctx.roundRect(8, 8, 240, 64, 12); else ctx.rect(8, 8, 240, 64);
  ctx.fill();
  ctx.fillStyle = "#ff8866";
  ctx.font = "bold 36px Arial";
  ctx.textAlign = "center";
  ctx.textBaseline = "middle";
  ctx.fillText("FRONT", 128, 42);
  const texture = new THREE.CanvasTexture(canvas);
  const spriteMat = new THREE.SpriteMaterial({ map: texture, transparent: true, depthTest: false });
  const sprite = new THREE.Sprite(spriteMat);
  sprite.position.copy(labelPos);
  sprite.scale.set(4.5, 1.5, 1);
  sprite.name = 'crosshatch_label';
  crosshatchGroup.add(sprite);

  // Other face labels - dynamic for all faces including subdivided ones
  const baseColors = ["#88aacc", "#88ccaa", "#ccaa88", "#aa88cc", "#cc8888", "#88cccc", "#cccc88", "#cc88cc", "#aabb88", "#88aaaa"];
  const faceLabelConfig = [];
  for (let i = 1; i < faces.length; i++) {
    faceLabelConfig.push({
      faceIdx: i,
      text: faces[i].name.toUpperCase(),
      color: baseColors[(i - 1) % baseColors.length],
    });
  }

  faceLabelConfig.forEach(cfg => {
    if (cfg.faceIdx >= faces.length) return;
    const f = faces[cfg.faceIdx];
    const fv = f.verts.map(i => cubeVertices[i]);
    const fc = new THREE.Vector3();
    fv.forEach(vv => fc.add(vv));
    fc.divideScalar(fv.length);
    const fn = getOutwardNormal(cfg.faceIdx);
    const lp = fc.clone().add(fn.clone().multiplyScalar(0.12));

    const lCanvas = document.createElement("canvas");
    lCanvas.width = 256; lCanvas.height = 64;
    const lCtx = lCanvas.getContext("2d");
    lCtx.fillStyle = "rgba(0,0,0,0.35)";
    lCtx.beginPath();
    if (lCtx.roundRect) lCtx.roundRect(16, 8, 224, 48, 10);
    else lCtx.rect(16, 8, 224, 48);
    lCtx.fill();
    lCtx.strokeStyle = cfg.color + "44";
    lCtx.lineWidth = 1.5;
    lCtx.beginPath();
    if (lCtx.roundRect) lCtx.roundRect(16, 8, 224, 48, 10);
    else lCtx.rect(16, 8, 224, 48);
    lCtx.stroke();
    lCtx.fillStyle = cfg.color;
    lCtx.globalAlpha = 0.7;
    lCtx.font = "bold 26px Arial";
    lCtx.textAlign = "center";
    lCtx.textBaseline = "middle";
    lCtx.fillText(cfg.text, 128, 34);

    const lTexture = new THREE.CanvasTexture(lCanvas);
    const lSpriteMat = new THREE.SpriteMaterial({
      map: lTexture, transparent: true, depthTest: false, opacity: 0.75,
    });
    const lSprite = new THREE.Sprite(lSpriteMat);
    lSprite.position.copy(lp);
    lSprite.scale.set(3.5, 1.0, 1);
    lSprite.name = `faceLabel_${cfg.faceIdx}`;
    crosshatchGroup.add(lSprite);
  });
}

// --- Volume / Area Calc ---
function signedVolumeOfTriangle(p1, p2, p3) {
  return p1.dot(p2.clone().cross(p3)) / 6.0;
}

function calculateOuterVolume() {
  let vol = 0;
  faces.forEach((face, fi) => {
    const v = face.verts.map(i => cubeVertices[i]);
    const normal = getFaceNormal(fi);
    const tris = triangulateFace(v, normal);
    tris.forEach(([a, b, c]) => {
      vol += signedVolumeOfTriangle(v[a], v[b], v[c]);
    });
  });
  return Math.abs(vol);
}

function calculateInnerVolume() {
  const innerVerts = cubeVertices.map((v, i) => {
    const normals = [];
    faces.forEach((face, fi) => {
      if (face.verts.includes(i)) {
        normals.push(getFaceNormal(fi).clone());
      }
    });
    const avgNormal = new THREE.Vector3();
    normals.forEach(n => avgNormal.add(n));
    avgNormal.normalize();
    const center = new THREE.Vector3();
    cubeVertices.forEach(cv => center.add(cv));
    center.divideScalar(cubeVertices.length);
    if (avgNormal.dot(center.clone().sub(v).normalize()) < 0) avgNormal.negate();
    return v.clone().add(avgNormal.multiplyScalar(WALL_THICKNESS));
  });
  let vol = 0;
  faces.forEach((face, fi) => {
    const v = face.verts.map(i => innerVerts[i]);
    const normal = getFaceNormal(fi);
    const tris = triangulateFace(v, normal);
    tris.forEach(([a, b, c]) => {
      vol += signedVolumeOfTriangle(v[a], v[b], v[c]);
    });
  });
  return Math.abs(vol);
}

function calculatePortDisplacement() {
  let totalDisp = 0;
  ports.forEach(p => {
    const r = p.diameter / 2;
    const tubeLen = p.tubeLength || DEFAULT_PORT_LENGTH;
    if (p.type === 'bassreflex') {
      totalDisp += Math.PI * r * r * tubeLen;
    } else {
      const magR = r * 0.45;
      totalDisp += Math.PI * magR * magR * tubeLen;
    }
  });
  return totalDisp;
}

function getNetInternalVolume() {
  return Math.max(0, calculateInnerVolume() - calculatePortDisplacement() - calculateBraceVolume());
}

function calculateFaceArea(faceVerts) {
  let area = 0;
  for (let i = 1; i < faceVerts.length - 1; i++) {
    const cross = faceVerts[i].clone().sub(faceVerts[0]).cross(faceVerts[i+1].clone().sub(faceVerts[0]));
    area += cross.length() / 2;
  }
  return area;
}

// --- Thiele-Small Calculations ---
function calcTuningFrequency() {
  const c = 13504;
  const netVol = getNetInternalVolume();
  if (netVol <= 0) return null;
  let totalPortArea = 0;
  let hasPort = false;
  ports.forEach(p => {
    if (p.type === 'bassreflex') {
      totalPortArea += Math.PI * Math.pow(p.diameter / 2, 2);
      hasPort = true;
    }
  });
  if (!hasPort) return null;
  const avgPortLen = ports.filter(p => p.type === 'bassreflex')
    .reduce((s, p) => s + (p.tubeLength || DEFAULT_PORT_LENGTH), 0)
    / ports.filter(p => p.type === 'bassreflex').length;
  const Leff = avgPortLen + 0.825 * Math.sqrt(totalPortArea);
  if (Leff <= 0) return null;
  const fb = (c / (2 * Math.PI)) * Math.sqrt(totalPortArea / (netVol * Leff));
  return fb;
}

function calcIdealPortLength(targetFb) {
  const c = 13504;
  const netVol = getNetInternalVolume();
  if (netVol <= 0 || targetFb <= 0) return null;
  let totalPortArea = 0;
  let hasPort = false;
  ports.forEach(p => {
    if (p.type === 'bassreflex') {
      totalPortArea += Math.PI * Math.pow(p.diameter / 2, 2);
      hasPort = true;
    }
  });
  if (!hasPort || totalPortArea <= 0) return null;
  const Leff = (totalPortArea * c * c) / (4 * Math.PI * Math.PI * targetFb * targetFb * netVol);
  const correction = 0.825 * Math.sqrt(totalPortArea);
  const portLen = Leff - correction;
  return portLen > 0.5 ? portLen : null;
}

function calcRecommendedVolume() {
  const sealedVb = tsParams.vas * Math.pow(tsParams.qts, 2.87);
  const portedVb = 15 * tsParams.vas * Math.pow(tsParams.qts, 3.3);
  return {
    sealed: sealedVb * 16.387,
    ported: portedVb * 16.387,
    sealedL: sealedVb,
    portedL: portedVb,
  };
}

function calcTargetF3() {
  return tsParams.fs * 0.9;
}

function isPortLengthOutOfRange(length) {
  return length < 3 || length > 10;
}

// =============== SWIPE GESTURE SYSTEM ===============
const SWIPE_THRESHOLD = 40; // minimum px distance to trigger swipe
const SWIPE_MAX_HORIZONTAL = 80; // max horizontal drift allowed

function addSwipeGesture(panel, { onSwipeUp, onSwipeDown }) {
  let startY = null, startX = null, startTime = 0;
  let tracking = false;

  panel.addEventListener('touchstart', (e) => {
    if (e.touches.length !== 1) return;
    // Don't interfere with inputs, selects, buttons
    const tag = e.target.tagName;
    if (tag === 'INPUT' || tag === 'SELECT' || tag === 'BUTTON') return;
    startY = e.touches[0].clientY;
    startX = e.touches[0].clientX;
    startTime = Date.now();
    tracking = true;
  }, { passive: true });

  panel.addEventListener('touchmove', (e) => {
    // passive — just track
  }, { passive: true });

  panel.addEventListener('touchend', (e) => {
    if (!tracking || startY === null) { tracking = false; return; }
    const elapsed = Date.now() - startTime;
    if (elapsed > 600) { tracking = false; startY = null; return; } // too slow

    let endY, endX;
    if (e.changedTouches && e.changedTouches.length > 0) {
      endY = e.changedTouches[0].clientY;
      endX = e.changedTouches[0].clientX;
    } else {
      tracking = false; startY = null; return;
    }

    const dy = endY - startY;
    const dx = Math.abs(endX - startX);
    tracking = false;
    startY = null;

    // Must be primarily vertical
    if (dx > SWIPE_MAX_HORIZONTAL) return;

    if (dy > SWIPE_THRESHOLD && onSwipeDown) {
      onSwipeDown();
    } else if (dy < -SWIPE_THRESHOLD && onSwipeUp) {
      onSwipeUp();
    }
  }, { passive: true });

  panel.addEventListener('touchcancel', () => { tracking = false; startY = null; });
}

function updateSubdivideUI() {
  // Check if any face has 4+ verts (prerequisite for subdivision)
  const canSubdivide = faces.some(f => f.verts.length >= 4);

  let statusText = '';
  if (subdivideMode) {
    if (subdivFirstVertex !== null) {
      statusText = `<span style="color:#44ff44;">1st vertex selected</span> — tap a <span style="color:#ffff44;">yellow</span> vertex to complete`;
    } else {
      statusText = `Tap the <span style="color:#44ff44;">1st vertex</span>, then a non-adjacent vertex on the same face`;
    }
  }

  subdivPanel.innerHTML = `
    <div style="display:flex;align-items:center;gap:8px;margin-bottom:${subdivideMode ? '6' : '0'}px;">
      <span style="font-size:10px;color:#88ccaa;flex-shrink:0;">✂ Subdivide Face</span>
      <div id="subdivSwitch" style="width:42px;height:22px;border-radius:11px;cursor:${canSubdivide ? 'pointer' : 'not-allowed'};background:${subdivideMode ? '#44ff44' : '#555'};position:relative;transition:background 0.2s;display:flex;align-items:center;opacity:${canSubdivide ? '1' : '0.4'};">
        <div style="width:18px;height:18px;border-radius:50%;background:#fff;position:absolute;left:${subdivideMode ? '22px' : '2px'};transition:left 0.2s;box-shadow:0 1px 4px rgba(0,0,0,0.3);"></div>
      </div>
      <span style="font-size:10px;color:${subdivideMode ? '#44ff44' : '#888'};">${subdivideMode ? 'ON' : 'OFF'}</span>
    </div>
    ${subdivideMode ? `
      <div style="font-size:9px;color:#88ccaa;line-height:1.4;padding:4px 0;">
        ${statusText}
      </div>
      ${subdivFirstVertex !== null ? `
        <div id="subdivCancelBtn" style="margin-top:4px;text-align:center;padding:5px 10px;border-radius:6px;border:1px solid rgba(255,68,68,0.3);background:rgba(255,68,68,0.1);color:#ff6666;font-size:10px;cursor:pointer;transition:all 0.2s;">
          ✕ Cancel Selection
        </div>
      ` : ''}
    ` : `
      <div style="font-size:8px;color:#667;margin-top:2px;">Split a face into two panels by connecting two non-adjacent vertices</div>
    `}
  `;

  const sw = document.getElementById('subdivSwitch');
  if (sw && canSubdivide) {
    const toggleSubdiv = (e) => {
      e.preventDefault(); e.stopPropagation();
      subdivideMode = !subdivideMode;
      if (!subdivideMode) {
        cancelSubdivide();
      } else {
        subdivFirstVertex = null;
        subdivEligibleVerts = [];
      }
      updateSubdivideUI();
    };
    sw.addEventListener('click', toggleSubdiv);
    sw.addEventListener('touchend', toggleSubdiv);
  }

  const cancelBtn = document.getElementById('subdivCancelBtn');
  if (cancelBtn) {
    const h = (e) => {
      e.preventDefault(); e.stopPropagation();
      subdivFirstVertex = null;
      subdivEligibleVerts = [];
      handles.forEach(h => {
        const isSplit = splitVertices.has(h.userData.vertexIndex);
        if (h.material && h.material.color) h.material.color.setHex(isSplit ? 0xff44ff : 0x00ccff);
        if (h.material && h.material.emissive) h.material.emissive.setHex(isSplit ? 0x661166 : 0x004466);
        h.scale.set(1, 1, 1);
      });
      updateSubdivideUI();
    };
    cancelBtn.addEventListener('click', h);
    cancelBtn.addEventListener('touchend', h);
  }
}

// Reverse-calculate port tube length from a target Fb
function calcPortLengthFromFb(targetFb) {
  const c = 13504;
  const netVol = getNetInternalVolume();
  if (netVol <= 0 || targetFb <= 0) return null;
  let totalPortArea = 0;
  let hasPort = false;
  ports.forEach(p => {
    if (p.type === 'bassreflex') {
      totalPortArea += Math.PI * Math.pow(p.diameter / 2, 2);
      hasPort = true;
    }
  });
  if (!hasPort || totalPortArea <= 0) return null;
  const Leff = (totalPortArea * c * c) / (4 * Math.PI * Math.PI * targetFb * targetFb * netVol);
  const correction = 0.825 * Math.sqrt(totalPortArea);
  const portLen = Leff - correction;
  return portLen > 0.1 ? portLen : null;
}

}

    });

      if (tsFocusedInputId === 'tsManualTubeLen') {
        applyTSPendingAndRefresh();
      }
    });
  }

  // --- Set to Ideal button ---
  const idealBtn = document.getElementById('tsSetIdealBtn');
  if (idealBtn) {
    const idealHandler = (e) => {
      e.preventDefault();
      e.stopPropagation();

      const idealLen = calcIdealPortLength(targetF3);
      if (idealLen !== null && idealLen > 0) {
        // Set all reflex ports to ideal length
        ports.forEach(p => {
          if (p.type === 'bassreflex') p.tubeLength = idealLen;
        });
        // Update the tube length input
        const tubeLenEl = document.getElementById('tsManualTubeLen');
        if (tubeLenEl) tubeLenEl.value = fmtTubeLenForInput(idealLen);
        // Recalculate Fb and update
        const newFb = calcTuningFrequency();
        const fbInputEl = document.getElementById('tsManualFb');
        if (fbInputEl && newFb !== null) fbInputEl.value = newFb.toFixed(1);
        // Flash the button green
        idealBtn.style.background = 'rgba(68,255,170,0.5)';
        idealBtn.style.color = '#fff';
        setTimeout(() => {
          idealBtn.style.background = 'rgba(68,255,170,0.15)';
          idealBtn.style.color = '#44ffaa';
        }, 400);
        buildPortVisuals();
        updateStats();
        updatePortPanel();
        // Deferred full TS rebuild (avoid killing focused input)
        setTimeout(() => updateTSPanel(), 500);
      }
    };
    idealBtn.addEventListener('click', idealHandler);
    idealBtn.addEventListener('touchend', idealHandler);
  }

  // --- Done button forces blur and full refresh ---
  const doneBtn = document.getElementById('tsDoneBtn');
  if (doneBtn) {
    const doneHandler = (e) => {
      e.preventDefault();
      e.stopPropagation();
      // Blur any focused input
      if (document.activeElement && document.activeElement.tagName === 'INPUT') {
        document.activeElement.blur();
      }
      tsFocusedInputId = null;
      tsPendingValues = {};
      updateTSPanel();
      updateStats();
      buildPortVisuals();
      updatePortPanel();
    };
    doneBtn.addEventListener('click', doneHandler);
    doneBtn.addEventListener('touchend', doneHandler);
  }
}

// =============== EDGE SPLIT INTERACTION ===============
// Detects clicks on edges (not handles) and splits the edge at that point

let edgeSplitPending = false; // flag to differentiate quick tap from drag

function tryEdgeSplit(pxy) {
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);

  // First check we're not clicking a handle or port drag handle
  const handleHits = raycaster.intersectObjects(handles);
  if (handleHits.length > 0) return false;
  const portHits = raycaster.intersectObjects(portDragHandles);
  if (portHits.length > 0) return false;

  // Check edge hit meshes
  const edgeHits = raycaster.intersectObjects(edgeHitMeshes);
  if (edgeHits.length === 0) return false;

  const hit = edgeHits[0];
  const edgeIdx = hit.object.userData.edgeIndex;
  if (edgeIdx === undefined) return false;

  const [a, b] = edges[edgeIdx];
  const va = cubeVertices[a];
  const vb = cubeVertices[b];
  const edgeLen = va.distanceTo(vb);

  // Don't split edges shorter than 2 inches
  if (edgeLen < 2) return false;

  // Use the hit point as the split position
  const splitPoint = hit.point.clone();

  // Verify the point is actually close to the edge line
  const edgeDir = vb.clone().sub(va).normalize();
  const toHit = splitPoint.clone().sub(va);
  const proj = va.clone().add(edgeDir.clone().multiplyScalar(toHit.dot(edgeDir)));
  const dist = splitPoint.distanceTo(proj);
  if (dist > EDGE_SPLIT_THRESHOLD * 1.5) return false;

  // Save state for undo, then perform the split
  saveUndoState();
  const newIdx = splitEdge(edgeIdx, splitPoint);

  // Flash effect: briefly show a bright sphere at split point
  showSplitFlash(cubeVertices[newIdx]);

  // Rebuild everything
  updateAll();
  updateTSPanel();

  return true;
}

// Visual flash when edge is split
function showSplitFlash(position) {
  const flashGeom = new THREE.SphereGeometry(1.2, 16, 16);
  const flashMat = new THREE.MeshBasicMaterial({
    color: 0xff44ff, transparent: true, opacity: 0.8,
  });
  const flash = new THREE.Mesh(flashGeom, flashMat);
  flash.position.copy(position);
  scene.add(flash);

  let t = 0;
  const fadeOut = () => {
    t += 0.04;
    flash.scale.setScalar(1 + t * 2);
    flashMat.opacity = 0.8 * (1 - t);
    if (t < 1) {
      requestAnimationFrame(fadeOut);
    } else {
      scene.remove(flash);
      flashGeom.dispose();
      flashMat.dispose();
    }
  };
  fadeOut();
}

// =============== TOUCH / INTERACTION ===============
function getPointerXY(event) {
  const rect = renderer.domElement.getBoundingClientRect();
  let clientX, clientY;
  if (event.touches && event.touches.length > 0) {
    clientX = event.touches[0].clientX; clientY = event.touches[0].clientY;
  } else if (event.changedTouches && event.changedTouches.length > 0) {
    clientX = event.changedTouches[0].clientX; clientY = event.changedTouches[0].clientY;
  } else {
    clientX = event.clientX; clientY = event.clientY;
  }
  return { x: ((clientX - rect.left) / rect.width) * 2 - 1, y: -((clientY - rect.top) / rect.height) * 2 + 1, clientX, clientY };
}

function tryStartPortDrag(pxy) {
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const hits = raycaster.intersectObjects(portDragHandles);
  if (hits.length > 0) {
    const handle = hits[0].object;
    const pi = handle.userData.portIndex;
    if (pi === undefined || pi < 0 || pi >= ports.length) return false;
    saveUndoState();
    draggingPort = pi;
    controls.enabled = false;
    const outNormal = getOutwardNormal(ports[pi].faceIndex);
    const worldPos = faceUVToWorld(ports[pi].faceIndex, ports[pi].u, ports[pi].v);
    // Use the face's actual plane for dragging (coplanar with face center)
    const face = faces[ports[pi].faceIndex];
    const faceCtr = new THREE.Vector3();
    face.verts.forEach(vi => faceCtr.add(cubeVertices[vi]));
    faceCtr.divideScalar(face.verts.length);
    draggingPortPlane.setFromNormalAndCoplanarPoint(outNormal, faceCtr);
    const planeHit = new THREE.Vector3();
    const hitResult = raycaster.ray.intersectPlane(draggingPortPlane, planeHit);
    if (hitResult !== null) {
      draggingPortOffset.set(planeHit.x - worldPos.x, planeHit.y - worldPos.y, planeHit.z - worldPos.z);
    } else {
      draggingPortOffset.set(0, 0, 0);
    }
    handle.material.emissive.setHex(0xffffff);
    handle.scale.set(1.4, 1.4, 1.4);
    return true;
  }
  return false;
}

function doPortDrag(pxy) {
  if (draggingPort === null) return;
  const port = ports[draggingPort];
  if (!port) return;
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const planeHit = new THREE.Vector3();
  const hitResult = raycaster.ray.intersectPlane(draggingPortPlane, planeHit);
  if (hitResult === null) return;
  const worldTarget = new THREE.Vector3(planeHit.x - draggingPortOffset.x, planeHit.y - draggingPortOffset.y, planeHit.z - draggingPortOffset.z);
  const uv = worldToFaceUV(port.faceIndex, worldTarget);
  const basis = getFaceBasis(port.faceIndex);
  const radiusU = (port.diameter / 2) / basis.uLen;
  const radiusV = (port.diameter / 2) / basis.vLen;
  // Clamp to the face bounding box with margin for port radius
  port.u = THREE.MathUtils.clamp(uv.u, radiusU + 0.02, 1 - radiusU - 0.02);
  port.v = THREE.MathUtils.clamp(uv.v, radiusV + 0.02, 1 - radiusV - 0.02);
  // Additional check: ensure the port center is inside the actual face polygon
  const testWorld = faceUVToWorld(port.faceIndex, port.u, port.v);
  if (!isPointInsideFace(port.faceIndex, testWorld)) {
    // Push toward face centroid
    const face = faces[port.faceIndex];
    const centroid = new THREE.Vector3();
    face.verts.forEach(vi => centroid.add(cubeVertices[vi]));
    centroid.divideScalar(face.verts.length);
    const centUV = worldToFaceUV(port.faceIndex, centroid);
    // Lerp toward centroid until inside
    for (let lerp = 0.1; lerp <= 1.0; lerp += 0.1) {
      const tryU = port.u + (centUV.u - port.u) * lerp;
      const tryV = port.v + (centUV.v - port.v) * lerp;
      const tryWorld = faceUVToWorld(port.faceIndex, tryU, tryV);
      if (isPointInsideFace(port.faceIndex, tryWorld)) {
        port.u = tryU;
        port.v = tryV;
        break;
      }
    }
  }
  buildPortVisuals();
}

// Check if a 3D point lies inside a face polygon (projected onto face plane)
function isPointInsideFace(faceIndex, point) {
  const face = faces[faceIndex];
  const basis = getFaceBasis(faceIndex);
  // Project point and face verts into 2D
  const pu = point.dot(basis.uDir);
  const pv = point.dot(basis.vDir);
  const poly2D = face.verts.map(i => {
    const p = cubeVertices[i];
    return { u: p.dot(basis.uDir), v: p.dot(basis.vDir) };
  });
  // Ray casting algorithm for point-in-polygon
  let inside = false;
  for (let i = 0, j = poly2D.length - 1; i < poly2D.length; j = i++) {
    const xi = poly2D[i].u, yi = poly2D[i].v;
    const xj = poly2D[j].u, yj = poly2D[j].v;
    if (((yi > pv) !== (yj > pv)) && (pu < (xj - xi) * (pv - yi) / (yj - yi) + xi)) {
      inside = !inside;
    }
  }
  return inside;
}

function tryStartDrag(pxy) {
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const intersects = raycaster.intersectObjects(handles);
  if (intersects.length > 0) {
    saveUndoState();
    controls.enabled = false;
    isDragging = true;
    selectedHandle = intersects[0].object;
    if (selectedHandle.material && selectedHandle.material.color) selectedHandle.material.color.setHex(0xff6644);
    if (selectedHandle.material && selectedHandle.material.emissive) selectedHandle.material.emissive.setHex(0xcc3300);
    selectedHandle.scale.set(1.5, 1.5, 1.5);
    const camDir = camera.getWorldDirection(new THREE.Vector3());
    dragPlane.setFromNormalAndCoplanarPoint(camDir.negate(), selectedHandle.position);
    const planeHit = new THREE.Vector3();
    const hit = raycaster.ray.intersectPlane(dragPlane, planeHit);
    if (hit !== null) {
      dragOffset.set(planeHit.x - selectedHandle.position.x, planeHit.y - selectedHandle.position.y, planeHit.z - selectedHandle.position.z);
    } else {
      dragOffset.set(0, 0, 0);
    }
    return true;
  }
  return false;
}

function doDrag(pxy) {
  if (!isDragging || !selectedHandle) return;
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);
  const planeHit = new THREE.Vector3();
  const hitResult = raycaster.ray.intersectPlane(dragPlane, planeHit);
  if (hitResult === null) return;
  const idx = selectedHandle.userData.vertexIndex;
  if (idx === undefined || idx < 0 || idx >= cubeVertices.length) return;
  const newX = snapValue(planeHit.x - dragOffset.x);
  const newY = snapValue(Math.max(0, planeHit.y - dragOffset.y));
  const newZ = snapValue(planeHit.z - dragOffset.z);
  const target = new THREE.Vector3(
    THREE.MathUtils.clamp(newX, -MAX_DIM / 2, MAX_DIM / 2),
    THREE.MathUtils.clamp(newY, 0, MAX_DIM),
    THREE.MathUtils.clamp(newZ, -MAX_DIM / 2, MAX_DIM / 2)
  );
  const oldPos = cubeVertices[idx].clone();
  const delta = target.clone().sub(oldPos);
  cubeVertices[idx].copy(target);
  if (symmetrical && idx < 8) {
    const mirrors = getSymmetryMap(idx);
    if (mirrors[0] !== undefined) {
      const m = cubeVertices[mirrors[0]];
      m.x = snapValue(-target.x);
      m.y = snapValue(THREE.MathUtils.clamp(m.y + delta.y, 0, MAX_DIM));
      m.z = snapValue(THREE.MathUtils.clamp(m.z + delta.z, -MAX_DIM / 2, MAX_DIM / 2));
    }
    if (mirrors[1] !== undefined) {
      const m = cubeVertices[mirrors[1]];
      m.x = snapValue(THREE.MathUtils.clamp(m.x + delta.x, -MAX_DIM / 2, MAX_DIM / 2));
      m.y = snapValue(THREE.MathUtils.clamp(m.y + delta.y, 0, MAX_DIM));
      m.z = snapValue(-target.z);
    }
    if (mirrors[2] !== undefined) {
      const m = cubeVertices[mirrors[2]];
      m.x = snapValue(-target.x);
      m.y = snapValue(THREE.MathUtils.clamp(m.y + delta.y, 0, MAX_DIM));
      m.z = snapValue(-target.z);
    }
  }
  updateAll();
  updateTSPanel();
}

function endDrag() {
  if (isDragging) {
    isDragging = false;
    if (selectedHandle) {
      const isSplit = splitVertices.has(selectedHandle.userData.vertexIndex);
      if (selectedHandle.material && selectedHandle.material.color) selectedHandle.material.color.setHex(isSplit ? 0xff44ff : 0x00ccff);
      if (selectedHandle.material && selectedHandle.material.emissive) selectedHandle.material.emissive.setHex(isSplit ? 0x661166 : 0x004466);
      selectedHandle.scale.set(1, 1, 1);
      selectedHandle = null;
    }
    controls.enabled = true;
  }
}

  if (touchStartPos) {
    const dx = pxy.clientX - touchStartPos.x;
    const dy = pxy.clientY - touchStartPos.y;
    if (Math.sqrt(dx*dx + dy*dy) > 10) touchStartPos = null;
  }

  if (draggingPort !== null) { endPortDrag(); e.preventDefault(); return; }
  if (isDragging) { endDrag(); e.preventDefault(); return; }

  // Check for quick tap
  const elapsed = Date.now() - touchStartTime;
  if (touchStartPos && elapsed < 400) {
    const pxy = getPointerXY(e);
    if (subdivideMode) {
      if (trySubdivideClick(pxy)) { e.preventDefault(); }
    } else {
      if (tryEdgeSplit(pxy)) { e.preventDefault(); }
    }
  }
  touchStartPos = null;
}, { passive: false });

renderer.domElement.addEventListener('mousemove', (e) => {
  const pxy = getPointerXY(e);
  if (draggingPort !== null) { doPortDrag(pxy); return; }
  if (isDragging) { doDrag(pxy); return; }

  // If moved significantly, cancel edge split
  if (mouseDownPos) {
    const dx = pxy.clientX - mouseDownPos.x;
    const dy = pxy.clientY - mouseDownPos.y;
    if (Math.sqrt(dx*dx + dy*dy) > 5) mouseDownPos = null;
  }

  // Hover
  touchMouse.set(pxy.x, pxy.y);
  raycaster.setFromCamera(touchMouse, camera);

  // Port handle hover
  const portHits = raycaster.intersectObjects(portDragHandles);
  portDragHandles.forEach(h => {
    if (h.material) { h.scale.set(1, 1, 1); h.material.opacity = 0.7; }
  });
  if (portHits.length > 0) {
    const h = portHits[0].object;
    h.scale.set(1.3, 1.3, 1.3);
    h.material.opacity = 1;
    renderer.domElement.style.cursor = "grab";
    return;
  }

  // Vertex handle hover
  const handleHits = raycaster.intersectObjects(handles);
  handles.forEach(h => {
    if (h !== selectedHandle && h.material) {
      const isSplit = splitVertices.has(h.userData.vertexIndex);
      if (h.material.color) h.material.color.setHex(isSplit ? 0xff44ff : 0x00ccff);
      if (h.material.emissive) h.material.emissive.setHex(isSplit ? 0x661166 : 0x004466);
      h.scale.set(1, 1, 1);
    }
  });
  if (handleHits.length > 0) {
    const hovered = handleHits[0].object;
    if (hovered.material && hovered.material.color) hovered.material.color.setHex(0x44ffaa);
    if (hovered.material && hovered.material.emissive) hovered.material.emissive.setHex(0x22aa66);
    hovered.scale.set(1.3, 1.3, 1.3);
    renderer.domElement.style.cursor = "grab";
    return;
  }

  // Subdivide mode hover
  if (subdivideMode) {
    updateSubdividePreview(pxy);
    // Show pointer cursor on eligible verts
    const subdivHits = raycaster.intersectObjects(handles);
    if (subdivHits.length > 0) {
      const vi = subdivHits[0].object.userData.vertexIndex;
      if (subdivFirstVertex === null) {
        // Any vertex on a face with 4+ verts
        const facesWithVert = getFacesForVertex(vi);
        const hasEligibleFace = facesWithVert.some(fi => faces[fi].verts.length >= 4);
        renderer.domElement.style.cursor = hasEligibleFace ? "pointer" : "not-allowed";
      } else {
        renderer.domElement.style.cursor = subdivEligibleVerts.includes(vi) ? "pointer" : "not-allowed";
      }
      return;
    }
    renderer.domElement.style.cursor = "default";
    return;
  }

  // Edge hover - show crosshair cursor when hovering an edge
  const edgeHits = raycaster.intersectObjects(edgeHitMeshes);
  if (edgeHits.length > 0) {
    renderer.domElement.style.cursor = "crosshair";
    return;
  }

  renderer.domElement.style.cursor = "default";
});

renderer.domElement.addEventListener('mouseup', (e) => {
  if (draggingPort !== null) { endPortDrag(); return; }

  // Check for quick click
  const elapsed = Date.now() - mouseDownTime;
  if (!isDragging && mouseDownPos && elapsed < 400) {
    const pxy = getPointerXY(e);
    const dx = pxy.clientX - mouseDownPos.x;
    const dy = pxy.clientY - mouseDownPos.y;

      // Subdivide mode takes priority
      if (subdivideMode) {
        trySubdivideClick(pxy);
      } else {
        tryEdgeSplit(pxy);
      }
    }
  }

  endDrag();
  mouseDownPos = null;
});

// --- Animate ---
const clock = new THREE.Clock();
function animate() {
  const t = clock.getElapsedTime();
  handles.forEach((h, i) => {
    if (h !== selectedHandle && h.children[0]) {
      const isSplit = splitVertices.has(h.userData.vertexIndex);
      const speed = isSplit ? 3 : 2;
      const pulse = 1 + Math.sin(t * speed + i * 0.8) * 0.08;
      h.children[0].scale.set(2.5 * pulse, 2.5 * pulse, 1);
    }
  });
  portGroup.children.forEach(child => {
    if (child.name && (child.name.startsWith('portGlow_') || child.name.startsWith('spkGlow_'))) {
      if (child.material) child.material.opacity = 0.25 + Math.sin(t * 3) * 0.12;
    }
  });
  controls.update();
  renderer.render(scene, camera);
}
